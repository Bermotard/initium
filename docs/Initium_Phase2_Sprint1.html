OBJECTIF

Implémenter l'architecture core Rust et la logique de lancement multi-plateforme

  • US-003: Lancer un site web au clic

1.4 Intégration Autostart OS

Tâche DEV-004: Enregistrement Système

  • Linux: XDG autostart (.desktop file)

  • Windows: Registry HKLM\Run

  • macOS: LaunchAgents (.plist)

// src/system/autostart.rs use cfg_if::cfg_if; pub struct AutostartManager; impl AutostartManager { pub fn enable() -> Result<(), String> { cfg_if! { if #[cfg(target_os = "linux")] { Self::enable_linux() } else if #[cfg(target_os = "windows")] { Self::enable_windows() } else if #[cfg(target_os = "macos")] { Self::enable_macos() } } } #[cfg(target_os = "linux")] fn enable_linux() -> Result<(), String> { let desktop_file = r#"[Desktop Entry] Type=Application Name=Initium Exec=/usr/local/bin/initium Autostart=true "#; let path = dirs::config_dir() .ok_or("Config dir not found")?; let autostart_path = path.join("autostart/initium.desktop"); std::fs::write(&autostart_path, desktop_file) .map_err(|e| e.to_string())?; log::info!("Initium registered for autostart on Linux"); Ok(()) } }

// src/config.rs use serde::{Deserialize, Serialize}; use std::path::Path; #[derive(Debug, Serialize, Deserialize, Clone)] pub struct Config { pub version: String, pub theme: String, pub autostart: bool, pub launchers: Vec<Launcher>, } impl Config { pub fn load<P: AsRef<Path>>(path: P) -> Result<Self, Box<dyn std::error::Error>> { let content = std::fs::read_to_string(path)?; let config = serde_json::from_str(&content)?; Ok(config) } pub fn save<P: AsRef<Path>>(&self, path: P) -> Result<(), Box<dyn std::error::Error>> { let json = serde_json::to_string_pretty(&self)?; std::fs::write(path, json)?; Ok(()) } pub fn add_launcher(&mut self, launcher: Launcher) { self.launchers.push(launcher); } pub fn remove_launcher(&mut self, id: &str) { self.launchers.retain(|l| l.id != id); } }
