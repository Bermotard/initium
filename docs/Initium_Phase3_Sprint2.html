// modules/api.js const API_BASE = 'http://127.0.0.1:8080/api'; class APIClient { async getLaunchers() { try { const response = await fetch(`${API_BASE}/launchers`); if (!response.ok) throw new Error('Failed to fetch launchers'); return await response.json(); } catch (error) { console.error('Error fetching launchers:', error); throw error; } } async addLauncher(launcher) { try { const response = await fetch(`${API_BASE}/add-launcher`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(launcher) }); if (!response.ok) throw new Error('Failed to add launcher'); return await response.json(); } catch (error) { console.error('Error adding launcher:', error); throw error; } } async removeLauncher(id) { try { const response = await fetch(`${API_BASE}/launcher/${id}`, { method: 'DELETE' }); if (!response.ok) throw new Error('Failed to delete launcher'); return await response.json(); } catch (error) { console.error('Error removing launcher:', error); throw error; } } async executeLauncher(id) { try { const response = await fetch(`${API_BASE}/execute`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ id }) }); if (!response.ok) throw new Error('Failed to execute launcher'); return await response.json(); } catch (error) { console.error('Error executing launcher:', error); throw error; } } } const api = new APIClient();
